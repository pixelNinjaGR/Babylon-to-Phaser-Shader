// Vertex shader
#if defined(WEBGL2) || defined(WEBGPU)
precision highp sampler2DArray;
#endif
precision highp float;

//Attributes
attribute vec3 position;
#ifdef UV1
attribute vec2 uv;
#else
vec2 uv = vec2(0.);
#endif


//Uniforms
uniform mat4 u_World;
uniform mat4 u_ViewProjection;
uniform vec3 u_Color;
uniform vec3 u_Color1;
uniform vec2 u_Speed_Noise;
uniform float u_time;
#ifdef UVTRANSFORM0
uniform mat4 textureTransform;
#endif
uniform vec2 u_Distortion;
uniform vec2 u_Speed;
uniform float u_time1;
#ifdef UVTRANSFORM1
uniform mat4 textureTransform1;
#endif


//Samplers
uniform  sampler2D ImageSourceBlock; 


//Varyings
#ifdef UVTRANSFORM0
varying vec2 transformedUV;
#endif
#ifdef VMAINOUTPUT2
varying vec2 vMainoutput2;
#endif
#ifdef UVTRANSFORM1
varying vec2 transformedUV1;
#endif
#ifdef VMAINOUTPUT4
varying vec2 vMainoutput4;
#endif




//Entry point
void main(void) {

//WorldPos
vec4 output1 = u_World * vec4(position, 1.0);

//WorldPos * ViewProjectionTransform
vec4 output0 = u_ViewProjection * output1;

//VertexOutput
gl_Position = output0;

//
vec2 output3 = uv;

//Panner
vec2 output2 = uv + u_Speed_Noise * u_time;

//ImageSourceBlock

//Texture
#ifdef UVTRANSFORM0
transformedUV = vec2(textureTransform * vec4(output2.xy, 1.0, 0.0));
#elif defined(VMAINOUTPUT2)
vMainoutput2 = output2.xy;
#endif
#ifdef UVTRANSFORM0
vec4 tempTextureRead = texture2D(ImageSourceBlock, transformedUV);
#elif defined(VMAINOUTPUT2)
vec4 tempTextureRead = texture2D(ImageSourceBlock, vMainoutput2);
#endif
float g = tempTextureRead.g;
#ifdef ISGAMMA0
                g = toLinearSpace(g);
                #endif
            
//
float output7 = g;

//Multiply
vec2 output6 = u_Distortion * output7;

//Add
vec2 output5 = output3 + output6;

//Panner
vec2 output4 = output5 + u_Speed * u_time1;

//Texture
#ifdef UVTRANSFORM1
transformedUV1 = vec2(textureTransform1 * vec4(output4.xy, 1.0, 0.0));
#elif defined(VMAINOUTPUT4)
vMainoutput4 = output4.xy;
#endif

}

// Fragment shader
#if defined(PREPASS)
#extension GL_EXT_draw_buffers : require
layout(location = 0) out highp vec4 glFragData[SCENE_MRT_COUNT];
highp vec4 gl_FragColor;
#endif
#if defined(WEBGL2) || defined(WEBGPU)
precision highp sampler2DArray;
#endif
precision highp float;

//Uniforms
uniform mat4 u_World;
uniform mat4 u_ViewProjection;
uniform vec3 u_Color;
uniform vec3 u_Color1;
uniform vec2 u_Speed_Noise;
uniform float u_time;
#ifdef UVTRANSFORM0
uniform mat4 textureTransform;
#endif
uniform vec2 u_Distortion;
uniform vec2 u_Speed;
uniform float u_time1;
#ifdef UVTRANSFORM1
uniform mat4 textureTransform1;
#endif
uniform float textureInfoName;
uniform float textureInfoName1;


//Samplers
uniform  sampler2D ImageSourceBlock; 


//Varyings
#ifdef UVTRANSFORM0
varying vec2 transformedUV;
#endif
#ifdef VMAINOUTPUT2
varying vec2 vMainoutput2;
#endif
#ifdef UVTRANSFORM1
varying vec2 transformedUV1;
#endif
#ifdef VMAINOUTPUT4
varying vec2 vMainoutput4;
#endif


//Texture
#include<helperFunctions>



//Entry point
void main(void) {

//ImageSourceBlock

//Texture
#ifdef UVTRANSFORM0
vec4 tempTextureRead2 = texture2D(ImageSourceBlock, transformedUV);
#elif defined(VMAINOUTPUT2)
vec4 tempTextureRead2 = texture2D(ImageSourceBlock, vMainoutput2);
#endif
float g = tempTextureRead2.g * textureInfoName;
#ifdef ISGAMMA2
                g = toLinearSpace(g);
                #endif
            
//Lerp
vec3 output9 = mix(u_Color , u_Color1, g);

//
vec3 output8 = output9;

//Texture
#ifdef UVTRANSFORM1
vec4 tempTextureRead3 = texture2D(ImageSourceBlock, transformedUV1);
#elif defined(VMAINOUTPUT4)
vec4 tempTextureRead3 = texture2D(ImageSourceBlock, vMainoutput4);
#endif
float r1 = tempTextureRead3.r * textureInfoName1;
#ifdef ISGAMMA3
                r1 = toLinearSpace(r1);
                #endif
            
//FragmentOutput
gl_FragColor  = vec4(output8, r1);
#ifdef CONVERTTOLINEAR0
gl_FragColor  = toLinearSpace(gl_FragColor);
#endif
#ifdef CONVERTTOGAMMA0
gl_FragColor  = toGammaSpace(gl_FragColor);
#endif
#if defined(PREPASS)
gl_FragData[0] = gl_FragColor;
#endif

}
